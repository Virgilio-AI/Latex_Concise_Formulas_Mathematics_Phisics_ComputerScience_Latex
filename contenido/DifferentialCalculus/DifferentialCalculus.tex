\begin{section}{Diferential Calculus}
$$(\tan(x))' = \sec ^2(x),\; (\csc (x))' = \csc(x)cot(x)$$
$$(\sec (x))' = \sec (x) \tan (x) ,\; (\cot (x))' = -\csc^2 (x)$$
$$(\ln (x))' = \frac{1}{x}$$
$$(a^{x})' = a^{x}ln(a)*x'$$
$$(\frac{f(x)}{g(x)})' = \frac{g(x)f(x)' = g(x)'f(x)}{(g(x))^2}$$


\end{section}
