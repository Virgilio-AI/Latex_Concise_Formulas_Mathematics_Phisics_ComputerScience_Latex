\begin{section}{Boolean Algebra}
	\begin{subsection}{Simple Formulas}
		$AA'=0, \; \; A+A' = 1 $
		$$AB + AC = A(B+C)$$
		$$(AB ... Z)' = A' + B' + ... + Z'$$
		$$(A+B)(A+C) = A + BC $$
		$$AB + AB' = A$$
		$$(A+B)(A+C) = A + BC $$
		$$A + A + B = A , \; \; A(A+D) = A$$
		$$(x+y)' = x'.y' $$
		$$(x.y)' = x' + y' $$
	\end{subsection}
\end{section}
